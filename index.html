<# Windows 11 Unattended Setup

This page is meant to be executed, not viewed.

Run from cmd:

curl -L ethanjansen.github.io/win11unattend | powershell -Command -

Source code is available at GitHub

#> try { Invoke-WebRequest https://ethanjansen.github.io/win11unattend/unattend.xml -OutFile "$env:WINDIR\Panther\unattend.xml" -ErrorAction Stop; Write-Host "Configuring Windows..."; $blv = Get-BitLockerVolume -MountPoint $env:SystemDrive -ErrorAction Stop; if ($blv.VolumeStatus -eq "FullyEncrypted" -or $blv.VolumeStatus -eq "EncryptionInProgress") { Disable-BitLocker -MountPoint $env:SystemDrive -ErrorAction Stop | Out-Null; do { Write-Host "Decrypting BitLocker: $($blv.EncryptionPercentage) encrypted..."; Start-Sleep -Seconds 10; $blv = Get-BitLockerVolume -MountPoint $env:SystemDrive; } while ($blv.VolumeStatus -ne "FullyDecrypted"); } Write-Host "Using /generalize"; & "$env:WINDIR\System32\Sysprep\Sysprep.exe" /generalize /oobe /unattend:"$env:WINDIR\Panther\unattend.xml" /reboot; } catch { Write-Error "Fatal error: $($_.Exception.Message)"; exit 1; } #EOF - This comment and the blank line above are VERY important for handling multi-line command blocks with powershell execution from stdin.
